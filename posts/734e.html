数据分析 

# _*_ coding: utf-8 _*_

from collections import Counter
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib
from time import *

time1 = time()

# 设置图片字体
matplotlib.rcParams['font.family'] = 'SimHei'
# 打开文件,(r)读取内容
f = open('D:/PyCharm/1211/file.txt', 'r')
# 将读取的内容存入变量
str = f.read()
# 格式化成字典
b = dict(Counter(str.split()))
# 显示重复次数超过50的人名
# print({key:value for key,value in b.items()if value > 50}) 
# 整理排序,(reverse=True)按从大到小排序
res = sorted(b.items(), key=lambda b: b[1], reverse=True)
# 取前十
df = pd.DataFrame(res[:10])
# 格式化打印格式
# df2 = pd.DataFrame({"名字": df[0], "次数": df[1]})
# 打印名字和重复次数
# print(df2)
# x,y轴的内容,以列表的形式赋值
x = list(df[0])
y = list(df[1])
# x,y轴的坐标系名称字体的大小的设定
plt.xlabel('名字', size=18)
plt.ylabel('重复次数', size=18)
# 设置x,y轴的条形图信息,宽度,颜色等
plt.bar(x, y, width=0.6, facecolor='orange', edgecolor='black')
# 传值,把x,y列表的'值'导入到坐标系
for a, b in zip(x, y):
    plt.text(a, b + 0.05, '%.0f' % b, ha='center', va='bottom', fontsize=10)
# 生成的图片信息
file_name = "重名次数统计图.jpg"
plt.title("重名次数统计图")
plt.savefig(file_name)
plt.show()
f.close()
time2 = time()
time3 = time2 - time1
print('该循环程序运行时间: ', time3)
