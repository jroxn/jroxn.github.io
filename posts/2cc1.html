class Wdatabases():
    def __init__(self):
        self.__db_host = "192.168.236.128"

        self.__db_user = "root"
        self.__db_password = "YjUzZjg0NTU1OWViM2I5ZTM1ZWE0NWZj"
        self.__db_database = "school"

    # 链接数据库
    def Wlink(self):
        self.__db = pymysql.connect(
            host=self.__db_host,
            port=self.__db_port,
            user=self.__db_user,
            password=self.__db_password,
            database=self.__db_database,
            charset='utf8'
        )

    # 插入数据
    def Winsert(self, name, gender, ages, cclass, yuwen, shuxue, english, total):
        try:
            # 连接数据库
            self.Wlink()
            # 创建游标
            global cursor
            cursor = self.__db.cursor()
            # sql命令
            sql = "insert into student(name,gender,age,cclass,yuwen,shuxue,english, total) value(%s,%s,%s,%s,%s,%s," \
                  "%s,%s)"
            # 执行sql命令
            cursor.execute(sql, (name, gender, ages, cclass, yuwen, shuxue, english, total))
        except Exception as e:
            print(e)
        finally:
            # 关闭游标
            cursor.close()
            # 提交
            self.__db.commit()
            # 关闭数据库连接
            self.__db.close()

    # 数据生成姓名, 年龄, 等并调用数据插入方法
